function createCalculator(startValue) { 
  let value = startValue; 
  return { 
    sum: function (newValue) { 
      value += newValue; 
      return value; 
    }, 
    mult: function (newValue) { 
      value *= newValue;
      return value;
    },
    sub: function (newValue) {
      value -= newValue;
      return value;
    },
    div: function (newValue) {
      value /= newValue;
      return value; 
    }, 
  }; 
}
